    37    package ti.sysbios.rta;
    38    
    39    import xdc.runtime.Log;
    40    import xdc.runtime.Diags;
    41    
    42    /*! 
    43     *  ======== SupportLogger ========
    44     *  The SupportLogger is used to log the Timestamp frequency so that
    45     *  the RTA tool can convert the log event timestamps into seconds. This log
    46     *  occurs during module startup.
    47     */
    48    
    49    @ModuleStartup
    50    
    51    module SupportLogger {
    52    
    53        /*! Logged in Module_startup to record Timestamp freq. */ 
    54        config Log.Event LM_timestampFreq = {
    55            mask: Diags.USER1 | Diags.USER2,
    56            msg: "LM_timestampFreq: Timestamp Frequency hi: %d, lo: %d"
    57        };
    58        
    59        /*! 
    60         *  ======== createLogSize ========
    61         *  Size of SupportLogger LoggerBuf in records. Allow one record per
    62         *  Task, Swi, and Hwi instance in the system.
    63         */
    64        config Int createLogSize = 32; 
    65    }
